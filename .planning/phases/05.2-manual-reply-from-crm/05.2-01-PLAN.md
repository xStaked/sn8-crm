Add method after `listConversationMessages`:
```typescript
async sendMessage(conversationId: string, body: string) {
  const normalizedId = this.normalizeParticipantPhone(conversationId);
  const kapsoMessageId = await this.messagingService.sendText(normalizedId, body);

  const fromPhone = this.configService.get<string>('KAPSO_PHONE_NUMBER_ID') ?? '';

  const created = await this.prisma.message.create({
    data: {
      externalMessageId: kapsoMessageId,
      direction: 'outbound',
      fromPhone,
      toPhone: normalizedId,
      body,
      channel: 'whatsapp',
      rawPayload: { source: 'crm_manual_reply', conversationId: normalizedId },
    },
  });

  return {
    id: created.id,
    conversationId: normalizedId,
    direction: 'outbound' as const,
    body: created.body,
    createdAt: created.createdAt.toISOString(),
  };
}
```
