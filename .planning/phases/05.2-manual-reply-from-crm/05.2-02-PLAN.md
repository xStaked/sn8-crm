Add state and handler inside the `DetailPanel` component, after the hook calls:
```typescript
const [messageBody, setMessageBody] = useState("");
const [sending, setSending] = useState(false);

async function handleSend() {
  if (!messageBody.trim() || !selectedId || sending) return;
  setSending(true);
  try {
    const sent = await apiFetchJson<ConversationMessageDto>(
      `/conversations/${encodeURIComponent(selectedId)}/messages`,
      { method: "POST", body: JSON.stringify({ body: messageBody.trim() }) },
    );
    await mutate((current) => [...(current ?? []), sent], { revalidate: false });
    setMessageBody("");
  } catch {
    // Error is visible via the SWR error state on next poll;
    // for v1 no toast is needed — the message stays in the textarea so the user can retry.
  } finally {
    setSending(false);
  }
}
```

Add the compose area JSX as the LAST child of the outermost `<div className="flex h-full flex-col bg-background">`, AFTER the messages area (after the closing of the state conditional blocks, before the final `</div>`), but ONLY when a conversation is selected (i.e., inside the branch where `selectedConversation` exists). Place it after the closing of the state-conditional rendering block (after the ternary chain for loading/unauthorized/error/empty/ready states):

```tsx
{/* Compose area — only visible when conversation is selected */}
{selectedConversation && state !== "unauthorized" && (
  <div className="border-t border-border px-6 py-4">
    <div className="flex items-end gap-3">
      <textarea
        className="flex-1 resize-none rounded-lg border border-border bg-card px-4 py-3 text-sm text-foreground placeholder:text-muted-foreground focus:outline-none focus:ring-2 focus:ring-ring"
        placeholder="Escribe un mensaje..."
        rows={2}
        value={messageBody}
        onChange={(e) => setMessageBody(e.target.value)}
        onKeyDown={(e) => {
          if (e.key === "Enter" && !e.shiftKey) {
            e.preventDefault();
            void handleSend();
          }
        }}
        disabled={sending}
      />
      <Button
        size="icon"
        disabled={!messageBody.trim() || sending}
        onClick={() => void handleSend()}
        className="h-10 w-10 shrink-0"
      >
        <Send className="h-4 w-4" />
      </Button>
    </div>
  </div>
)}
```

IMPORTANT layout rule: The compose area must be a SIBLING of the messages scroll div, not nested inside it. The outer flex column structure is:

2. Messages/state div (flex-1 overflow-y-auto)
3. Compose div (border-t, fixed height, NO flex-grow)

This ensures the message scroll area shrinks to accommodate the compose area without hiding messages behind it.
